package classFrame;

import java.io.File;

import org.openqa.selenium.By;

import org.openqa.selenium.firefox.FirefoxDriver;

import org.testng.annotations.Test;

import jxl.Cell;
import jxl.Sheet;
import jxl.Workbook;
import jxl.read.biff.BiffException;

public class DataDrivenFrame 
{	
	@Test(dataProvider="MyData")
	public void Excel(String email,String pass) throws InterruptedException
	{
		WebDriver d=new FirefoxDriver();
		Thread.sleep(5000);
		d.get("http://www.facebook.com");
		Thread.sleep(5000);
		d.findElement(By.id("email")).sendKeys(email);
		Thread.sleep(5000);
		d.findElement(By.id("pass")).sendKeys(pass);
		Thread.sleep(5000);
		d.findElement(By.id("loginbutton")).click();
		Thread.sleep(5000);
		d.close();
		
	}
	@DataProvider(name="MyData")
	public String[][] Contents() throws BiffException, IOException
	{
		System.out.println(" Enter File Name");
		Scanner sc=new Scanner(System.in);
		String s1=sc.next();
		File f=new File(s1);
		//File f=new File("?C:\\Users\\prash\\Documents\\New folder\\DataFrame\\yuppy.xls");
		Workbook wb=Workbook.getWorkbook(f);
		Sheet sh=wb.getSheet(0);
		int row=sh.getRows();
		int col=sh.getColumns();
		String data[][]=new String[row][col];
		for(int i=0;i<row;i++)
		{
			
			for(int j=0;j<col;j++)
			{
				
				Cell values=sh.getCell(j, i);
				data[i][j]=values.getContents();
			}
		}
		return data;
		
	}

}

package simple;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;
import org.openqa.selenium.firefox.FirefoxDriver;
import org.openqa.selenium.ie.InternetExplorerDriver;

public class TestCaseProcess 
{
	WebDriver d;
	
	public void Open_Browser(String tdata)
	{
		if(tdata.equalsIgnoreCase("Firefox"))
		{
			
			d=new FirefoxDriver();
			
		}
		else if(tdata.equalsIgnoreCase("Chrome"))
		{
			System.setProperty("", "");
			d=new ChromeDriver();
			
		}
		else if(tdata.equalsIgnoreCase("IE"))
		{
			System.setProperty("", "");
			d=new InternetExplorerDriver();
			
		}
		
	}
	public void Enter_Url(String tdata)
	{
		d.get(tdata);
		
	}
	public void Enter_Text(String ptype,String pvalue,String tdata)
	{
		By b=ElementValues(ptype,pvalue);
		d.findElement(b).sendKeys(tdata);
		
	}
	public void Click_Button(String ptype,String pvalue)
	{
		By b=ElementValues(ptype,pvalue);
		d.findElement(b).click();
		
	}
	
	public By ElementValues(String ptype,String pvalue)
	{
		By b;
		switch(ptype)
		{
			case "id":
				b=By.id(pvalue);
				break;
				
			case "name":
				b=By.name(pvalue);
				break;
				
			case "cssSelector":
				b=By.cssSelector(pvalue);
				break;
				
			case "linkText":
				b=By.linkText(pvalue);
				break;
				
			case "partialLinkText":
				b=By.partialLinkText(pvalue);
				break;
				
			case "tagName":
				b=By.tagName(pvalue);
				break;
				
			case "xpath":
				b=By.xpath(pvalue);
				break;
				
			default:
				b=null;
				break;	
		}
		return b;
		
	}

}

package simple;

import java.io.FileInputStream;
import java.io.FileNotFoundException;
import java.io.IOException;

import jxl.Sheet;
import jxl.Workbook;
import jxl.read.biff.BiffException;

public class SheetHandling 
{
	
	Workbook wb;
	Sheet sh;
	
	public void ExcelSheet(String Filepath) throws BiffException, IOException
	{
		FileInputStream fs=new FileInputStream(Filepath);
		wb=Workbook.getWorkbook(fs);
		sh=wb.getSheet(0);
		
	}
	
	public String GetCellValues(int c,int r)
	{
		String data=sh.getCell(c, r).getContents();
		return data;
		
	}

	public int Rows()
	{
		
		int r=sh.getRows();
		return r;
	}
	
	public int Cols()
	{
		int c=sh.getColumns();
		return c;
		
	}
}

package simple;

import java.io.IOException;
import java.util.Scanner;

import org.testng.annotations.Test;

import jxl.read.biff.BiffException;

public class ExecutionProcess 
{
	@Test(priority=1)
	public void FinalStep() throws BiffException, IOException, InterruptedException
	{
		
		SheetHandling sheet=new SheetHandling();
		System.out.println(" Enter The File Here ");
		Scanner sc=new Scanner(System.in);
		String s1=sc.next();
		sheet.ExcelSheet(s1);
		int rcount=sheet.Rows();
		int ccount=sheet.Cols();
		System.out.println(" Rows "+rcount);
		System.out.println(" Columns "+ccount);
		String tname="";
		String ptype="";
		String pvalue="";
		String tdata="";
		TestCaseProcess t1=new TestCaseProcess();
		for(int r=1;r<rcount;r++)
		{
			
			tname=sheet.GetCellValues(0, r);
			ptype=sheet.GetCellValues(1, r);
			pvalue=sheet.GetCellValues(2, r);
			tdata=sheet.GetCellValues(3, r);
			
			System.out.println(tname+" "+ptype+" "+pvalue+" "+tdata);
			Thread.sleep(6000);
			switch(tname)
			{
			
				case "Open_Browser":
					t1.Open_Browser(tdata);
					break;
				
				case "Enter_Url":
					t1.Enter_Url(tdata);
					break;
					
				case "Enter_Text":
					t1.Enter_Text(ptype, pvalue, tdata);
					break;
				
				case "Click_Button":
					t1.Click_Button(ptype, pvalue);
					break;
					
				default:
					System.out.println(" No TestCase ");
					break;
			}
		}
		
	}

}

Working On the SoapUI Ng:
---------------------------------

1: Following the Tutorials on SoapUI Ng
2: Process the input and Output formats by XML and JSon.
3: Create New Project
	a: Provide the Project Name
	b: Provide the URL where the WebServices to be Tested.
	c: At End of the URL always enter " ?wsdl" .
	d: click on next and finish.
	e: Create the TestSuite
		1: Create a TestCase (We can create a Separate TestCase for each Operation or We can Setup single TestCase fro all the Operations)
		2: Create the TestSteps.
	f: Click on the Build button and provide the DataSource Values.
	g: Configure the data once the data is entered into the Data Source.
	h: Then Hit Back Button and run the task for the particular testcase to be performed.
