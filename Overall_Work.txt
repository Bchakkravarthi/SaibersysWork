Passing values from one TestCase to another TestCase by using the String of Arrays concept:

package sampleSkit;

import org.testng.annotations.DataProvider;
import org.testng.annotations.Test;

public class PassingTestcases 
{
	@Test(dataProvider="my values")
	public void Digit(String x,String y)
	{
		
		int z=Integer.parseInt(x)+Integer.parseInt(y);
		System.out.println(" Sum of two numbers is "+z);
		
	}
	@DataProvider(name="my values")
	public String[][] Data()
	{
		
		int a=19;
		int b=28;
		String[][] s=new String[1][2];
		
		s[0][0]=Integer.toString(a);
		
		s[0][1]=Integer.toString(b);
		
		return s;
	}

}

Handling the Mouse_Action functions:


package sampleSkit;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.WebElement;
import org.openqa.selenium.firefox.FirefoxDriver;
import org.openqa.selenium.interactions.Actions;

public class ActionOnMouse 
{

	public static void main(String[] args) throws InterruptedException 
	{
		
		WebDriver f=new FirefoxDriver();
		Thread.sleep(5000);
		f.get("http://store.demoqa.com");
		Thread.sleep(5000);
		WebElement s=f.findElement(By.xpath(".//*[@id='menu-item-33']/a"));
		Actions a=new Actions(f);
		a.moveToElement(s).build().perform();
		f.findElement(By.xpath(".//*[@id='menu-item-35']/a")).click();
		Thread.sleep(5000);
		f.close();
		

	}

}
