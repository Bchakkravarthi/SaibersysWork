Data Driven Automation_Testing:
---------------------------------------

In Data_Driven_Automation_Testing multiple Tests are performed by different Data.

Procedure:
-------------

1: Create a simple FacebookLogin Class()
	1. Open the Browser.
	2. Open Facebook Page.
	3. SignIn into the Account( Valid UserId and Valid Password).
	4. Click on the DP image and proceed to Logout.
	5. Close the Browser.

These is the general procedure for testing a Facebook_Login_Logout_Page, But when it comes to Data_Driven_Automation_Testing we are able the same webpage with morethan "One set of data".

Here We pass the set of data using the EXCEL SHEET.

	1: Download the Excel sheet from microsoft office.
	2: Download the JXL JAR files 2.4.2 
	3: Always import the JXL JAR files after the creation of package.
	4: If you are using the 2016 Excel Sheet make sure You save the file by change the "Excel Workbook" ---> "Excel 97 2003 Workbook".

Example:
-----------

package automation;

import java.io.File;
import java.io.IOException;

import org.openqa.selenium.By;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.firefox.FirefoxDriver;
import org.testng.annotations.DataProvider;
import org.testng.annotations.Test;

import jxl.Cell;
import jxl.Sheet;
import jxl.Workbook;
import jxl.read.biff.BiffException;

public class AutomationData 
{
	@Test(dataProvider="My Data")
	public void FacebookLogin(String Email,String Pass) throws InterruptedException
	{
		WebDriver d=new FirefoxDriver();
		Thread.sleep(5000);
		d.get("http://www.facebook.com");
		Thread.sleep(5000);
		d.findElement(By.id("email")).sendKeys(Email);
		Thread.sleep(5000);
		d.findElement(By.id("pass")).sendKeys(Pass);
		Thread.sleep(5000);
		d.findElement(By.id("loginbutton")).click();
		Thread.sleep(5000);
		d.findElement(By.id("userNavigationLabel")).click();
		Thread.sleep(5000);
		d.findElement(By.xpath(".//*[@id='BLUE_BAR_ID_DO_NOT_USE']/div/div/div[1]/div/div/ul/li[12]/a/span/span")).click();
		Thread.sleep(5000);
		d.close();	
	}
	@DataProvider(name="My Data")
	public String[][] FacebookValues() throws BiffException, IOException
	{
		File f=new File("D:\\Facebook.xls");
		Workbook wb=Workbook.getWorkbook(f);
		Sheet s=wb.getSheet(0);
		int row=s.getRows();
		int col=s.getColumns();
		String data[][]=new String[row][col];
		for(int i=0;i<row;i++)
		{
			for(int j=0;j<col;j++)
			{
				Cell values=s.getCell(j,i);
				data[i][j]=values.getContents();
				
			}
		}
		return data;
	}

}
